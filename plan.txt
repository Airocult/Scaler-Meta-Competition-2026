from pydantic import BaseModel, Field
from typing import Literal, Optional, Any
from enum import Enum

class ServiceStatus(BaseModel):
    name: str
    status: Literal["healthy", "degraded", "down", "unknown"]
    error_rate: float  # 0.0–1.0
    latency_p99_ms: int
    restarts_last_hour: int

class IncidentPhase(str, Enum):
    INVESTIGATING = "investigating"
    MITIGATING = "mitigating"
    VERIFYING = "verifying"
    RESOLVED = "resolved"

class Action(BaseModel):
    action_type: Literal[
        "read_logs",
        "check_metrics",
        "list_services",
        "check_alerts",
        "check_deployments",
        "check_dependencies",
        "run_diagnostic",
        "apply_fix",
        "verify_health",
        "write_postmortem",
        "escalate"
    ]
    parameters: dict[str, Any] = Field(default_factory=dict)
    reasoning: str = Field(min_length=10)

class Observation(BaseModel):
    timestamp: str
    alert_summary: str
    service_statuses: dict[str, ServiceStatus]
    last_action_result: str
    incident_phase: IncidentPhase
    available_actions: list[str]
    step_count: int
    time_elapsed_minutes: int
    hints_used: int  # tracks if agent called escalate

class Reward(BaseModel):
    step_reward: float        # reward for this step (-0.1 to +0.3)
    cumulative_reward: float  # total so far
    episode_score: float      # final 0.0–1.0 (only meaningful when done=True)
    done: bool
    info: dict[str, Any]

class ResetRequest(BaseModel):
    task_id: str
    seed: Optional[int] = 42   # reproducibility default

class StepResponse(BaseModel):
    observation: Observation
    reward: Reward

class RewardShaper:
    """
    Converts raw scenario signals into step-level rewards.
    Ensures the reward function has meaningful gradient everywhere —
    not just binary at episode end.
    """

    BASE_STEP_PENALTY = -0.01   # small cost per step (time pressure)

    REWARDS = {
        # Positive
        "root_cause_progress":     +0.15,  # traced one hop toward root cause
        "root_cause_identified":   +0.30,  # correctly identified the exact cause
        "fix_applied_correctly":   +0.30,  # applied the right remediation
        "resolution_verified":     +0.15,  # confirmed system healthy after fix
        "postmortem_written":      +0.10,  # wrote meaningful postmortem
        "postmortem_quality":      +0.05,  # postmortem mentions root cause term

        # Negative
        "wrong_fix_applied":       -0.10,  # applied fix to wrong service/type
        "destructive_action":      -0.20,  # e.g., restart of healthy service
        "escalate_used":           -0.05,  # hint used (per use)
        "repeated_same_action":    -0.05,  # same action+params twice in a row
    }

    def compute(
        self,
        event: str,
        step_count: int,
        max_steps: int,
        previous_reward: float
    ) -> float:
        base = self.BASE_STEP_PENALTY
        event_reward = self.REWARDS.get(event, 0.0)
        return round(base + event_reward, 4)

class SREBenchEnvironment:
    """
    Stateful environment. One instance per active episode.
    Thread-safe via a lock — support concurrent API calls.
    """

    def __init__(self):
        self._scenario: Optional[BaseScenario] = None
        self._lock = asyncio.Lock()

    async def reset(self, task_id: str, seed: int = 42) -> Observation:
        async with self._lock:
            scenario_map = {
                "task1_memory_leak":    MemoryLeakScenario,
                "task2_db_cascade":     DBCascadeScenario,
                "task3_race_condition": RaceConditionScenario,
            }
            if task_id not in scenario_map:
                raise ValueError(f"Unknown task_id: {task_id}")
            self._scenario = scenario_map[task_id](seed=seed)
            return self._scenario.get_initial_observation()

    async def step(self, action: Action) -> tuple[Observation, Reward]:
        async with self._lock:
            if self._scenario is None:
                raise RuntimeError("Call reset() before step()")
            obs, step_reward, done = self._scenario.apply_action(action)
            reward = Reward(
                step_reward=step_reward,
                cumulative_reward=self._scenario.cumulative_reward,
                episode_score=self._scenario.get_grader_score() if done else 0.0,
                done=done,
                info={
                    "step_count": obs.step_count,
                    "incident_phase": obs.incident_phase,
                    "task_id": self._scenario.task_id,
                }
            )
            return obs, reward

    async def state(self) -> dict:
        if self._scenario is None:
            return {"status": "no_episode_active"}
        return self._scenario.get_current_state()

    async def get_grader_score(self) -> float:
        if self._scenario is None:
            raise RuntimeError("No active episode")
        return self._scenario.get_grader_score()

from fastapi import FastAPI, HTTPException
from contextlib import asynccontextmanager
from app.environment import SREBenchEnvironment
from app.models import *
import yaml, asyncio

env = SREBenchEnvironment()

@asynccontextmanager
async def lifespan(app: FastAPI):
    yield  # startup/shutdown hooks if needed

app = FastAPI(
    title="SREBench",
    description="On-Call Incident Response OpenEnv Environment",
    version="1.0.0",
    lifespan=lifespan
)

@app.post("/reset", response_model=Observation)
async def reset(request: ResetRequest):
    return await env.reset(request.task_id, request.seed)

@app.post("/step", response_model=StepResponse)
async def step(action: Action):
    obs, reward = await env.step(action)
    return StepResponse(observation=obs, reward=reward)

@app.get("/state")
async def state():
    return await env.state()

@app.get("/tasks")
async def tasks():
    """Returns task list + full action schema for agent consumption."""
    with open("openenv.yaml") as f:
        config = yaml.safe_load(f)
    return {
        "tasks": config["tasks"],
        "action_schema": config["action_space"],
        "observation_schema": config["observation_space"],
    }

@app.get("/grader")
async def grader():
    score = await env.get_grader_score()
    return {"episode_score": score}

@app.post("/baseline")
async def baseline():
    """
    Triggers the baseline inference script programmatically.
    Runs all 3 tasks with seed=42, returns scores.
    Heavy — uses thread pool to avoid blocking event loop.
    """
    import subprocess, sys, json
    result = subprocess.run(
        [sys.executable, "baseline/run_baseline.py", "--json"],
        capture_output=True, text=True, timeout=300
    )
    if result.returncode != 0:
        raise HTTPException(500, f"Baseline script failed: {result.stderr}")
    return json.loads(result.stdout)

@app.get("/health")
async def health():
    return {"status": "ok", "environment": "srebench", "version": "1.0.0"}

"""
Baseline agent: GPT-4o runs each task with ReAct-style prompting.
Reads OPENAI_API_KEY from environment.
Usage:
  python baseline/run_baseline.py            # prints human-readable
  python baseline/run_baseline.py --json     # prints JSON only (for /baseline endpoint)
"""

import os, sys, json, argparse, httpx
from openai import OpenAI

BASE_URL = os.getenv("SREBENCH_URL", "http://localhost:7860")
TASKS = ["task1_memory_leak", "task2_db_cascade", "task3_race_condition"]
SEED = 42
MODEL = "gpt-4o"

SYSTEM_PROMPT = """You are an expert SRE on-call engineer.
You will receive an incident alert and must investigate and resolve it.
On each turn, output a JSON action in this exact format:
{
  "action_type": "<one of the available actions>",
  "parameters": { ... },
  "reasoning": "Your step-by-step reasoning before acting"
}
Always think carefully about the dependency graph before applying fixes.
Prefer to read_logs and check_metrics before applying any fix.
"""

def run_episode(client: OpenAI, task_id: str) -> float:
    """Runs one full episode. Returns final grader score."""
    # 1. Reset
    reset_resp = httpx.post(f"{BASE_URL}/reset",
                            json={"task_id": task_id, "seed": SEED})
    obs = reset_resp.json()

    messages = [
        {"role": "system", "content": SYSTEM_PROMPT},
        {"role": "user", "content":
            f"Incident alert: {obs['alert_summary']}\n"
            f"Available actions: {obs['available_actions']}\n"
            f"Service statuses: {json.dumps(obs['service_statuses'], indent=2)}\n"
            "What do you do first?"}
    ]

    max_steps = {"task1_memory_leak": 20,
                 "task2_db_cascade": 30,
                 "task3_race_condition": 40}[task_id]

    for step_num in range(max_steps):
        # 2. Get LLM action
        completion = client.chat.completions.create(
            model=MODEL,
            messages=messages,
            response_format={"type": "json_object"},
            temperature=0.2,
        )
        raw_action = json.loads(completion.choices[0].message.content)

        # 3. Validate + send to env
        try:
            step_resp = httpx.post(f"{BASE_URL}/step",
                                   json=raw_action, timeout=30)
            step_data = step_resp.json()
        except Exception as e:
            break

        obs = step_data["observation"]
        reward = step_data["reward"]

        # 4. Add to message history
        messages.append({"role": "assistant",
                          "content": completion.choices[0].message.content})
        messages.append({"role": "user", "content":
            f"Result: {obs['last_action_result']}\n"
            f"Incident phase: {obs['incident_phase']}\n"
            f"Step reward: {reward['step_reward']}\n"
            f"Service statuses: {json.dumps(obs['service_statuses'], indent=2)}\n"
            + (f"Available actions: {obs['available_actions']}"
               if step_num % 3 == 0 else "")
        })

        if reward["done"]:
            return reward["episode_score"]

    # Episode ended without done=True — get grader score
    grader_resp = httpx.get(f"{BASE_URL}/grader")
    return grader_resp.json()["episode_score"]


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--json", action="store_true")
    args = parser.parse_args()

    client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])
    results = {}

    for task_id in TASKS:
        score = run_episode(client, task_id)
        results[task_id] = round(score, 4)

    if args.json:
        print(json.dumps(results))
    else:
        print("\n=== SREBench Baseline Results ===")
        for task_id, score in results.items():
            bar = "█" * int(score * 20)
            print(f"  {task_id:<28} {score:.4f}  {bar}")
        print(f"\n  Average: {sum(results.values())/len(results):.4f}")

if __name__ == "__main__":
    main()

"""
Scans all Python source files under app/ to ensure NO bare calls to
module-level random functions exist. All randomness must go through
a seeded Random or numpy Generator instance.
"""
import ast, pytest
from pathlib import Path

BANNED_CALLS = {
    ("random", "random"), ("random", "randint"), ("random", "choice"),
    ("random", "shuffle"), ("random", "seed"),
    ("np", "random"),    # Catches np.random.xxx calls at module level
}

def find_bare_random_calls(path: Path) -> list[str]:
    violations = []
    try:
        tree = ast.parse(path.read_text())
    except SyntaxError:
        return []
    for node in ast.walk(tree):
        if isinstance(node, ast.Call):
            func = node.func
            if isinstance(func, ast.Attribute):
                if isinstance(func.value, ast.Name):
                    key = (func.value.id, func.attr)
                    if key in BANNED_CALLS:
                        violations.append(
                            f"{path}:{node.lineno} — bare call: "
                            f"{func.value.id}.{func.attr}()"
                        )
    return violations

def test_no_bare_random_in_app():
    files = list(Path("app").rglob("*.py"))
    all_violations = []
    for f in files:
        all_violations.extend(find_bare_random_calls(f))
    assert not all_violations, (
        "Found unseeded random calls — wrap in Random(seed) or "
        "numpy default_rng(seed):\n" + "\n".join(all_violations)
    )

import pytest
from app.reward import RewardShaper
from eval.config import CFG

shaper = RewardShaper()

ALL_EVENTS = list(shaper.REWARDS.keys()) + ["unknown_event_xyz"]

@pytest.mark.parametrize("event", ALL_EVENTS)
def test_reward_for_every_event_in_bounds(event):

def test_base_step_penalty_is_negative():

"""
Executes a scripted action sequence and returns the final grader score.
Used by all oracle episode tests.
"""
import asyncio
from app.environment import SREBenchEnvironment
from app.models import Action

async def run_scripted_episode(
    task_id: str,
    actions: list[dict],
    seed: int = 42,
) -> dict:
    """
    Runs a fully scripted episode.
    Returns {score, done, steps_taken, cumulative_reward, final_phase}.
    """
    env = SREBenchEnvironment()
    await env.reset(task_id, seed=seed)
    cumulative = 0.0
    last_obs = None
    done = False

    for i, raw in enumerate(actions):
        action = Action(**raw)
        obs, reward = await env.step(action)
        cumulative = reward.cumulative_reward
        last_obs = obs
        done = reward.done
        if done:
            return {
                "score": reward.episode_score,
                "done": True,
                "steps_taken": i + 1,
                "cumulative_reward": cumulative,
                "final_phase": obs.incident_phase.value,
            }

    final_score = await env.get_grader_score()
    return {
        "score": final_score,
        "done": done,
        "steps_taken": len(actions),
        "cumulative_reward": cumulative,
        "final_phase": last_obs.incident_phase.value if last_obs else "unknown",
    }

class DimensionScore(BaseModel):
    score: float = Field(ge=0.0, le=10.0)

    scenario_realism: DimensionScore
    grader_fairness: DimensionScore
    observation_completeness: DimensionScore

    overall_verdict: Literal["pass", "warn", "fail"]
    top_issues: list[str]                # Top 3 most critical issues
    auto_patch_hints: list[str]          # Concrete code-level suggestions

"""
LLM judge that evaluates the SREBench environment quality across
4 dimensions. Returns structured critique JSON.
"""
import json, os
from pathlib import Path
from openai import OpenAI
from eval.layer5_meta.critique_schema import EnvironmentCritique, DimensionScore
from eval.config import CFG

PROMPTS_DIR = Path("eval/layer5_meta/prompts")

class MetaEvaluator:
    def __init__(self):
        self.client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

    def _build_context(self, task_id: str, episode_trajectory: list[dict]) -> str:
        """Serializes an episode trajectory for the LLM to review."""
        return json.dumps({
            "task_id": task_id,
            "trajectory": episode_trajectory[:15],  # cap at 15 steps for tokens
        }, indent=2)

    def _load_prompt(self, name: str) -> str:
        return (PROMPTS_DIR / f"{name}.txt").read_text()

    def evaluate(
        self,
        task_id: str,
        episode_trajectory: list[dict],
    ) -> EnvironmentCritique:
        context = self._build_context(task_id, episode_trajectory)

        master_prompt = f"""
You are an expert RL environment designer reviewing an OpenEnv environment
called SREBench (on-call SRE incident response simulation).

You will evaluate the environment across 4 dimensions and return structured
JSON matching this exact schema:
{{
  "evaluated_task": "<task_id>",
  "reward_signal_quality": {{
    "score": <0-10>,
    "verdict": "pass|warn|fail",
    "issues": ["..."],
    "suggestions": ["..."]
  }},
  "scenario_realism": {{...}},
  "grader_fairness": {{...}},
  "observation_completeness": {{...}},
  "overall_score": <0-10>,
  "overall_verdict": "pass|warn|fail",
  "top_issues": ["...", "...", "..."],
  "auto_patch_hints": ["...", "...", "..."]
}}

Scoring guide:
- reward_signal_quality: Does the reward function guide the agent to root cause?
  Are partial rewards meaningful? Is the signal dense enough?
  fail = only sparse reward, no gradient | warn = mostly ok but gaps | pass = good signal
- scenario_realism: Does this look like a real SRE incident?
  Are the logs/metrics convincing? Could a real engineer use this for training?
  fail = clearly fake/toy | warn = plausible but thin | pass = realistic
- grader_fairness: Can the grader be gamed? Are the scoring weights proportional
  to difficulty? Is there unintended hard cap?
  fail = trivially exploitable | warn = minor issues | pass = fair and robust
- observation_completeness: Does the agent have all the information it needs?
  Are hints given when appropriate? Is info withheld unfairly?
  fail = agent missing critical info | warn = some gaps | pass = complete

Here is the episode trajectory to evaluate:
{context}

Return ONLY the JSON object, no preamble, no markdown fences.
"""
        resp = self.client.chat.completions.create(
            model=CFG.meta_model,
            messages=[{"role": "user", "content": master_prompt}],
            temperature=CFG.meta_temperature,
            response_format={"type": "json_object"},
        )

        raw = json.loads(resp.choices[0].message.content)

        # Parse with Pydantic for strict validation
        critique = EnvironmentCritique(
            evaluated_task=raw["evaluated_task"],
            reward_signal_quality=DimensionScore(**raw["reward_signal_quality"]),
            scenario_realism=DimensionScore(**raw["scenario_realism"]),
            grader_fairness=DimensionScore(**raw["grader_fairness"]),
            observation_completeness=DimensionScore(**raw["observation_completeness"]),
            overall_score=raw["overall_score"],
            overall_verdict=raw["overall_verdict"],
            top_issues=raw.get("top_issues", []),
            auto_patch_hints=raw.get("auto_patch_hints", []),
        )
        return critique

"""
Compares current baseline scores to historical snapshots.
Alerts on unexpected drift (environment changed without intent).
"""
import json, os
from datetime import datetime
from pathlib import Path
from eval.config import CFG

SNAPSHOT_DIR = Path("eval/drift/baseline_snapshots")

class DriftDetector:
    def detect(self, current_scores: dict[str, float]) -> dict:
        """
        Compare current_scores to the most recent snapshot.
        Returns {status: ok|warn|fail, drifts: [...]}
        """
        snapshots = sorted(SNAPSHOT_DIR.glob("snapshot_*.json"))
        if not snapshots:
            self._save_snapshot(current_scores)
            return {"status": "ok", "drifts": [],
                    "note": "No prior snapshot. Saved current scores as baseline."}

        latest = json.loads(snapshots[-1].read_text())
        drifts = []

        for task_id, current in current_scores.items():
            prior = latest.get(task_id)
            if prior is None:
                continue
            delta = abs(current - prior)
            if delta >= CFG.drift_fail_threshold:
                drifts.append({
                    "task_id": task_id,
                    "prior": prior,
                    "current": current,
                    "delta": round(delta, 4),
                    "severity": "fail",
                    "message": (
                        f"HARD FAIL: Score drifted {delta:.3f} "
                        f"({prior:.3f} → {current:.3f})"
                    )
                })
            elif delta >= CFG.drift_alert_threshold:
                drifts.append({
                    "task_id": task_id,
                    "prior": prior,
                    "current": current,
                    "delta": round(delta, 4),
                    "severity": "warn",
                    "message": (
                        f"WARNING: Score drifted {delta:.3f} "
                        f"({prior:.3f} → {current:.3f})"
                    )
                })

        if any(d["severity"] == "fail" for d in drifts):
            status = "fail"
        elif drifts:
            status = "warn"
        else:
            status = "ok"
            self._save_snapshot(current_scores)  # Save new snapshot on clean run

        return {"status": status, "drifts": drifts}

    def _save_snapshot(self, scores: dict):
        SNAPSHOT_DIR.mkdir(parents=True, exist_ok=True)
        fname = SNAPSHOT_DIR / f"snapshot_{datetime.utcnow().strftime('%Y%m%d_%H%M%S')}.json"
        fname.write_text(json.dumps(scores, indent=2))

"""
Scans for trivial exploit paths: action sequences shorter than N steps
that achieve scores above the single_action_max_score threshold.
Runs after oracle episodes to check if any short path scored too high.
"""
import asyncio
from app.environment import SREBenchEnvironment
from app.models import Action
from eval.config import CFG

SHORT_PATH_THRESHOLD = 3   # Any ≤ 3 action sequence scoring above limit = exploit

CANDIDATE_EXPLOITS = [
    # One-shot write_postmortem with magic keywords
    [
        Action(action_type="write_postmortem",
               parameters={"content": "lock_timeout race condition rollback fixed"},
               reasoning="Trying to shortcut task3 with keywords in postmortem"),
    ],
    # Immediate rollback without any investigation on task3
    [
        Action(action_type="apply_fix",
               parameters={"service": "inventory-service",
                            "fix_type": "rollback_config",
                            "deploy_id": "deploy_001"},
               reasoning="Guessing deploy_id to skip investigation"),
    ],
    # Verify health immediately on task1
    [
        Action(action_type="verify_health",
               parameters={"service": "order-service"},
               reasoning="Verify before fixing — should not resolve"),
        Action(action_type="apply_fix",
               parameters={"service": "order-service", "fix_type": "restart"},
               reasoning="Apply fix"),
    ],
]

async def scan_exploits() -> list[dict]:
    found_exploits = []
    tasks_to_check = [
        ("task1_memory_leak", CANDIDATE_EXPLOITS[2]),
        ("task3_race_condition", CANDIDATE_EXPLOITS[0]),
        ("task3_race_condition", CANDIDATE_EXPLOITS[1]),
    ]

    for task_id, actions in tasks_to_check:
        env = SREBenchEnvironment()
        await env.reset(task_id, seed=42)
        final_score = 0.0
        for action in actions:
            _, reward = await env.step(action)
            if reward.done:
                final_score = reward.episode_score
                break
        else:
            final_score = await env.get_grader_score()

        if final_score > CFG.single_action_max_score:
            found_exploits.append({
                "task_id": task_id,
                "exploit_actions": [a.action_type for a in actions],
                "score": final_score,
                "threshold": CFG.single_action_max_score,
            })

    return found_exploits

def build_report(results: dict) -> dict:
    """
    results = {
      "layer1": {"passed": bool, "tests": int, "failures": list},
      "layer2": {"passed": bool, ...},
      "layer3": {"passed": bool, "episode_scores": dict},

    if overall_status == "pass" and results.get("drift", {}).get("status") == "warn":
        overall_status = "warn"

    report = {
        "generated_at": datetime.utcnow().isoformat() + "Z",

            "drift_status":             results.get("drift", {}).get("status"),
        }
    }

    Path("eval_report.json").write_text(json.dumps(report, indent=2))

#!/usr/bin/env python3
"""
Master evaluation runner for SREBench.
Executes all 5 layers + drift detection + report generation.

Usage:
  python eval/run_eval.py                    # Full eval, all layers
  python eval/run_eval.py --skip-meta        # Skip LLM meta (saves API cost)
  python eval/run_eval.py --layer 1 2 3      # Run specific layers only
  python eval/run_eval.py --save-snapshot    # Force-save drift snapshot
  python eval/run_eval.py --ci               # CI mode: exit 1 on any blocking fail
"""

import argparse, asyncio, json, subprocess, sys
from pathlib import Path

def run_pytest(test_path: str, extra_args: list[str] = None) -> dict:
    """Runs pytest on a path and returns {passed, tests, failures}."""
    cmd = [
        sys.executable, "-m", "pytest", test_path,
        "-v", "--tb=short", "--json-report",
        "--json-report-file=/tmp/pytest_report.json",
        "-q"
    ] + (extra_args or [])

    result = subprocess.run(cmd, capture_output=True, text=True)
    passed = result.returncode == 0

    try:
        report = json.loads(Path("/tmp/pytest_report.json").read_text())
        tests = report.get("summary", {}).get("total", 0)
        failures = [
            f"{t['nodeid']}: {t.get('call', {}).get('longrepr', '')[:200]}"
            for t in report.get("tests", [])
            if t["outcome"] == "failed"
        ]
    except Exception:
        tests = 0
        failures = [result.stdout[-500:]] if not passed else []

    return {"passed": passed, "tests": tests, "failures": failures}


async def run_meta_eval() -> dict:
    """Runs LLM meta-evaluation on all 3 tasks."""
    from eval.layer5_meta.meta_evaluator import MetaEvaluator
    from eval.layer3_oracle.runner import run_scripted_episode
    from eval.layer3_oracle.episodes import task1_optimal, task2_optimal, task3_optimal
    from eval.layer5_meta.issue_generator import generate_issues, save_issues

    evaluator = MetaEvaluator()
    critiques = []
    episodes = [
        ("task1_memory_leak", task1_optimal.ACTIONS),
        ("task2_db_cascade", task2_optimal.ACTIONS),
        ("task3_race_condition", task3_optimal.ACTIONS),
    ]

    for task_id, actions in episodes:
        # Run episode and collect trajectory
        trajectory = []
        from app.environment import SREBenchEnvironment
        from app.models import Action
        env = SREBenchEnvironment()
        await env.reset(task_id, seed=42)
        for raw in actions:
            obs, reward = await env.step(Action(**raw))
            trajectory.append({
                "action": raw,
                "observation": obs.model_dump(),
                "step_reward": reward.step_reward,
                "done": reward.done,
            })
            if reward.done:
                break

        critique = evaluator.evaluate(task_id, trajectory)
        critiques.append(critique)

    issues = generate_issues(critiques)
    save_issues(issues)

    avg_score = sum(c.overall_score for c in critiques) / len(critiques)
    any_fail = any(c.overall_verdict == "fail" for c in critiques)

    return {
        "passed": not any_fail,
        "avg_score": round(avg_score, 2),
        "critiques": [c.model_dump() for c in critiques],
    }


async def run_baseline_and_drift() -> tuple[dict, dict]:
    """Runs baseline scores and drift detection."""
    from eval.drift.drift_detector import DriftDetector
    from eval.drift.exploit_scanner import scan_exploits
    import httpx

    # Get current baseline scores from running env
    scores = {}
    async with httpx.AsyncClient(app=__import__("app.main", fromlist=["app"]).app,
                                  base_url="http://test") as client:
        for task_id in ["task1_memory_leak", "task2_db_cascade", "task3_race_condition"]:
            await client.post("/reset", json={"task_id": task_id, "seed": 42})
            score = (await client.get("/grader")).json()["episode_score"]
            scores[task_id] = score

    drift_result = DriftDetector().detect(scores)
    exploits = await scan_exploits()

    return (
        {
            "passed": drift_result["status"] != "fail" and not exploits,
            "drift": drift_result,
            "exploits": exploits,
            "baseline_scores": scores,
        },
        scores,
    )


def main():
    parser = argparse.ArgumentParser(description="SREBench Self-Evaluation Runner")
    parser.add_argument("--skip-meta", action="store_true",
                        help="Skip LLM meta-evaluation (saves API cost)")
    parser.add_argument("--layer", nargs="+", type=int,
                        help="Run only specific layers (1-5)")
    parser.add_argument("--ci", action="store_true",
                        help="Exit 1 if any blocking layer fails")
    parser.add_argument("--save-snapshot", action="store_true",
                        help="Force save current scores as drift snapshot")
    args = parser.parse_args()

    run_layers = set(args.layer or [1, 2, 3, 4, 5])
    results = {}

    if 1 in run_layers:
        print("Running Layer 1: Static analysis...")
        results["layer1"] = run_pytest("eval/layer1_static/")

    if 2 in run_layers:
        print("Running Layer 2: Unit + property tests...")
        results["layer2"] = run_pytest("eval/layer2_unit/")

    if 3 in run_layers:
        print("Running Layer 3: Oracle episodes...")
        results["layer3"] = run_pytest("eval/layer3_oracle/assertions.py")

    if 4 in run_layers:
        print("Running Layer 4: Adversarial probes...")
        results["layer4"] = run_pytest("eval/layer4_adversarial/")

    if 5 in run_layers and not args.skip_meta:
        print("Running Layer 5: LLM meta-evaluation...")
        results["layer5"] = asyncio.run(run_meta_eval())
    elif 5 in run_layers:
        print("Skipping Layer 5 (--skip-meta)")

    print("Running drift detection...")
    drift_result, baseline_scores = asyncio.run(run_baseline_and_drift())
    results["drift"] = drift_result
    results["baseline_scores"] = baseline_scores

    from eval.report import build_report
    report = build_report(results)

    if args.ci:
        from eval.config import CFG
        blocking_failed = any(
            not results.get(f"layer{l}", {}).get("passed", True)
            for l in [1, 2, 3, 4]
        )
        if blocking_failed:
            print("CI GATE FAILED — blocking layers have failures")
            sys.exit(1)
        if CFG.drift_blocks and drift_result.get("drift", {}).get("status") == "fail":
            print("CI GATE FAILED — score drift exceeds threshold")
            sys.exit(1)
        print("CI GATE PASSED")
        sys.exit(0)


if __name__ == "__main__":
    main()

"""
Reads eval_issues.json and generates concrete code patch suggestions
using the OpenAI API. These are NOT applied automatically — they are
saved to eval_patch_proposals.md for human review.

This closes the self-improvement loop:
  Eval → Critique → Issues → Patch Proposals → Human Review → Apply → Re-eval
"""
import json, os
from openai import OpenAI
from eval.config import CFG

def generate_patch_proposals(issues_path: str = "eval_issues.json") -> str:
    client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])
    issues = json.loads(open(issues_path).read())

    if not issues:
        return "No issues found. Environment is clean."

    issues_text = "\n\n".join(
        f"ISSUE {i+1}: {iss['title']}\n{iss['body'][:400]}"
        for i, iss in enumerate(issues[:5])  # Cap at 5 most critical
    )

    prompt = f"""
You are an expert RL environment engineer. Below are issues found in the
SREBench OpenEnv environment by an automated evaluator.

For each issue, provide:
1. A one-sentence diagnosis of the root cause in the code
2. The specific file and function that needs to change
3. A concrete code snippet showing the fix (10-30 lines max)

Issues:
{issues_text}

Format your response as Markdown with one section per issue.
Each section: ## Issue N — <title>
Then: **File**: `path/to/file.py`
Then: **Function**: `function_name()`
Then: A fenced Python code block with the fix.
"""

    resp = client.chat.completions.create(
        model=CFG.meta_model,
        messages=[{"role": "user", "content": prompt}],
        temperature=0.2,
    )

    proposals = f"# SREBench Auto-Patch Proposals\n\n" \
                f"Generated from {len(issues)} issues in eval_issues.json\n\n" \
                + resp.choices[0].message.content

    with open("eval_patch_proposals.md", "w") as f:
        f.write(proposals)

    print(f"Saved patch proposals to eval_patch_proposals.md")
    return proposals


if __name__ == "__main__":
    print(generate_patch_proposals())

2. Layer 3 oracle episodes must cover the FULL score range:
   - At least one episode per task scoring in each bucket:
     [0.0–0.30] [0.30–0.65] [0.65–1.00]
   This ensures the grader has meaningful gradient everywhere.

3. The drift snapshot system must use UTC timestamps only.
   Never use local time — CI runs may be in different time zones.

4. Layer 5 meta-eval must be idempotent: running it twice on the
   same trajectory must produce the same overall_verdict.
   (Temperature=0.1 is set for this reason.)

5. eval/run_eval.py --ci must exit 0 if all blocking layers pass,
   even if layer5 or drift produce warnings.
   exit 1 only for: blocking layer failures OR drift status="fail".

6. The GitHub Actions workflow must cache pip dependencies.
   First CI run ≤ 4 minutes with warm cache.

7. All pytest files must use pytest-asyncio with
   @pytest.mark.asyncio and asyncio_mode = "auto" in pytest.ini:

   [pytest]
   asyncio_mode = auto
   testpaths = tests eval

Why This System Catches Every Known Failure Mode
The 5-layer defence covers a different class of bug at each level — they don't overlap, they compound.
Layer 1 catches contract violations that would silently break openenv validate — untyped fields, missing yaml keys, bare random.choice() calls that make the environment non-reproducible.
Layer 2 catches mathematical invariants — a grader that accidentally returns 1.05, a reward shaper whose positive events sum past 1.2, or a scenario whose incident_phase can jump from investigating → resolved.
Layer 3 catches semantic correctness — the grader scores too much for too little work, or the optimal path inexplicably scores below 0.85. These are the bugs that pass all unit tests but make the environment useless for training.
Layer 4 catches active exploitation — single-action shortcuts, postmortem spam, keyword injection into write_postmortem content. Without this layer, a model could score 0.9 by writing "lock_timeout race condition rollback" in its very first postmortem call.
Layer 5 closes the feedback loop automatically — the LLM critique identifies qualitative issues (thin log templates, reward signal gaps, unrealistic cascades) that deterministic tests cannot, and converts them to concrete patch proposals in eval_patch_proposals.md rather than leaving them as vague complaints.
